<script>
    let tar = document.getElementById("target");
    function go(){
        let colors = [colorgen.randomRGB(), colorgen.randomRGB(), colorgen.randomRGB()];
        let alpha = Math.random();
        let blocks = tar.getElementsByClassName("b");
        for (let i = 0; i < blocks.length; i++) {
            let cssColor = colorgen.toCSSString(colors[i], i === 2 ? alpha : undefined);
            blocks[i].style.backgroundColor = cssColor;
        }
    }
    go();
    tar.onclick = go;
</script>
